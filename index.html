Java 

    static <T> Optional<T> lastOption(List<T> list) {
      if (list.isEmpty) return Optional.empty();
      else return Optional.of(list.get(list.size() - 1));
    }
Idris 

    lastOption : List a -> Maybe a
    lastOption       []  = Nothing
    lastOption (x :: []) = Just x
    lastOption (x :: xs) = lastOption xs
